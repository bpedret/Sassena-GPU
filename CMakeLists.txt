CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

SET(CMAKE_PROJECT_PATH ${CMAKE_PROJECT_PATH} "${CMAKE_SOURCE_DIR}/cmake")
SET(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${CMAKE_SOURCE_DIR}/cmake/modules")

ADD_DEFINITIONS( -fPIC -Wall)
SET(CMAKE_EXE_LINKER_FLAGS -fPIC)

PROJECT(Sassena)
#enable_language (Fortran)
INCLUDE(${CMAKE_PROJECT_PATH}/CMakeLists.txt.version)

SET(CMAKE_EXE_LINKER_FLAGS "-Wall ${CMAKE_EXE_LINKER_FLAGS}")

IF(DEBUG)
SET(CMAKE_BUILD_TYPE Debug)
ELSE(DEBUG)
SET(CMAKE_BUILD_TYPE Release)
ENDIF(DEBUG)

IF(STATIC)
INCLUDE(${CMAKE_PROJECT_PATH}/CMakeLists.txt.static)
ELSE(STATIC)
INCLUDE(${CMAKE_PROJECT_PATH}/CMakeLists.txt.shared)
ENDIF(STATIC)
